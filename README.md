# @materialios/icons
@materialios's icons

## Features
* IconWrapper
* ArrowBackIcon
* MenuIcon

## Installation
```s
npm i @materialios/icons
```

## Example
```c
import { MenuIcon } from '@materialios/icons'

const App = () => {
  return (
    <div className='App'>
      <MenuIcon>
    </div>
  )
}

export default App
```
